    message: string,
    public toolName: string,
    public originalError?: Error
  ) {
    super(message);
    this.name = 'FigmaToolError';
  }
}

export function handleToolError(
  error: unknown,
  toolName: string
): { content: Array<{ type: "text"; text: string }> } {
